SELECT DISTINCT
-- Columns
      CURRENT DATE AS VANDAAG, A.RJAPPC, A.RJPDLC, A.RJOBJC, A.RJOBJL, A.RJOBJT
     , A.RJOBJA, A.RJOBJD, B.ELROBC, B.ELROBL, B.ELROBT, B.ELRMBC, B.ELRELT,
      B.ELUSDF, COUNT(*)
-- Tables
      FROM "V60TST"/"OMOBJ" A,
           "V60TST"/"OMREL" B
-- Row Selection
      WHERE (A.RJOBJC = 'OMLOG')
-- Join Conditions
           AND (RJOBJC = ELOBJC)
           AND (RJOBJL = ELOBJL)
           AND (RJOBJT = ELOBJT)
-- Summary Columns
      GROUP BY A.RJAPPC, A.RJPDLC, A.RJOBJC, A.RJOBJL, A.RJOBJT, A.RJOBJA,
      A.RJOBJD, B.ELROBC, B.ELROBL, B.ELROBT, B.ELRMBC, B.ELRELT, B.ELUSDF
-- Sort Columns
      ORDER BY VANDAAG, A.RJAPPC, A.RJPDLC, A.RJOBJC, A.RJOBJL, A.RJOBJT
